Simple Example


import { praxys } from './src/index.ts';

const store = praxys({
  count: 0,
  user: {
    name: 'John',
    visits: 0
  }
});

// This watcher would normally depend on count
store.watch(() => {
  // Create dependency on store.count only
  console.log(`Count is: ${store.count}`);
  
  // Read user data without creating dependencies
  console.log(`User ${store.user.name} has visited ${store.user.visits} times`);
});

// This triggers the watcher because count changed
store.count = 1;

// This doesn't trigger the watcher because we accessed user.visits in ignore mode
store.ignore(() => {
  store.user.visits = 5;
});
